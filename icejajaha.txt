    local function updateDrag(input)
        local delta = input.Position - dragStart
        MainFrameShadow.Position = UDim2.new(startPosition.X.Scale, startPosition.X.Offset + delta.X, startPosition.Y.Scale, startPosition.Y.Offset + delta.Y)
    end
    MainFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            isDragging = true
            dragStart = input.Position
            startPosition = MainFrameShadow.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    isDragging = false
                end
            end)
        end
    end)
    MainFrame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if input == dragInput and isDragging then
            updateDrag(input)
        end
    end)

    function IceHub.CreateTab(tabName)
        local tabButton = Instance.new("TextButton")
        local tabButtonCorner = Instance.new("UICorner")
        local tabStroke = Instance.new("UIStroke")
        tabButton.Name = tabName .. "Tab"
        tabButton.Parent = tabScrollFrame
        tabButton.BackgroundColor3 = Color3.fromRGB(30, 30, 35)
        tabButton.BackgroundTransparency = 1
        tabButton.Size = UDim2.new(0, 130, 0, 32)
        tabButton.Font = Enum.Font.Gotham
        tabButton.Text = tabName
        tabButton.TextColor3 = Color3.fromRGB(200, 200, 200)
        tabButton.TextSize = 14
        tabButtonCorner.CornerRadius = UDim.new(0, 6)
        tabButtonCorner.Parent = tabButton
        tabStroke.Parent = tabButton
        tabStroke.Color = Color3.fromRGB(255, 255, 255)
        tabStroke.BackgroundTransparency = 1
        tabStroke.Thickness = 0
        local pageFrame = Instance.new("ScrollingFrame")
        local pageListLayout = Instance.new("UIListLayout")
        pageFrame.Name = tabName .. "Page"
        pageFrame.Parent = pageContainer
        pageFrame.Active = true
        pageFrame.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        pageFrame.BackgroundTransparency = 1
        pageFrame.BorderSizePixel = 0
        pageFrame.Size = UDim2.new(0, 342, 0, 233)
        pageFrame.CanvasSize = UDim2.new(0, 0, 0, 0)
        pageFrame.ScrollBarThickness = 3
        pageFrame.ScrollBarImageColor3 = Color3.fromRGB(80, 80, 80)
        pageFrame.Visible = false
        pageListLayout.Name = "pageListLayout"
        pageListLayout.Parent = pageFrame
        pageListLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
        pageListLayout.SortOrder = Enum.SortOrder.LayoutOrder
        pageListLayout.Padding = UDim.new(0, 9)
        table.insert(activePages, pageFrame)
        table.insert(tabButtonList, tabButton)
        if #activePages == 1 then
            pageFrame.Visible = true
            tabButton.BackgroundTransparency = 0
            tabButton.TextColor3 = Color3.fromRGB(0, 255, 150)
            tabStroke.BackgroundTransparency = 0.9
        end
        tabButton.MouseButton1Click:Connect(function()
            for _, page in pairs(activePages) do
                page.Visible = false
            end
            for _, btn in pairs(tabButtonList) do
                btn.BackgroundTransparency = 1
                btn.TextColor3 = Color3.fromRGB(200, 200, 200)
                if btn:ToTensor() and btn:FindFirstChild("UIStroke") then
                    btn.UIStroke.BackgroundTransparency = 1
                end
            end
            pageFrame.Visible = true
            tabButton.BackgroundTransparency = 0
            tabButton.TextColor3 = Color3.fromRGB(0, 255, 150)
            tabStroke.BackgroundTransparency = 0.9
        end)
        
        local pageElements = {}
        
        -- [تعديل أمير - تلوين دالة الأزرار بالتفصيل]
        function pageElements.CreateButton(text, callback)
            local buttonFrame = Instance.new("TextButton")
            local buttonCorner = Instance.new("UICorner")
            buttonFrame.Name = "Button"
            buttonFrame.Parent = pageFrame
            buttonFrame.Size = UDim2.new(0, 330, 0, 35)
            -- ألوان الزر الملكية
            buttonFrame.BackgroundColor3 = Color3.fromRGB(28, 28, 32)
            buttonFrame.Font = Enum.Font.Gotham
            buttonFrame.Text = text
            buttonFrame.TextColor3 = Color3.fromRGB(255, 255, 255)
            buttonFrame.TextSize = 14
            buttonCorner.CornerRadius = UDim.new(0, 6)
            buttonCorner.Parent = buttonFrame
            
            -- حواف حمراء لكل زر
            local bStroke = Instance.new("UIStroke", buttonFrame)
            bStroke.Color = Color3.fromRGB(255, 0, 50)
            bStroke.Thickness = 1
            
            buttonFrame.MouseButton1Click:Connect(function()
                pcall(callback)
            end)
        end
        
        -- [تعديل أمير - تلوين دالة التوجلات بالتفصيل]
        function pageElements.CreateToggle(text, default, callback)
            local toggled = default
            local toggleFrame = Instance.new("Frame")
            local toggleCorner = Instance.new("UICorner")
            local toggleText = Instance.new("TextLabel")
            local toggleButton = Instance.new("TextButton")
            local toggleDisplay = Instance.new("Frame")
            local displayCorner = Instance.new("UICorner")
            toggleFrame.Name = "ToggleFrame"
            toggleFrame.Parent = pageFrame
            toggleFrame.BackgroundColor3 = Color3.fromRGB(22, 22, 26)
            toggleFrame.Size = UDim2.new(0, 330, 0, 38)
            toggleCorner.CornerRadius = UDim.new(0, 6)
            toggleCorner.Parent = toggleFrame
            toggleText.Name = "ToggleText"
            toggleText.Parent = toggleFrame
            toggleText.BackgroundTransparency = 1
            toggleText.Position = UDim2.new(0, 14, 0, 0)
            toggleText.Size = UDim2.new(0, 200, 0, 38)
            toggleText.Font = Enum.Font.Gotham
            toggleText.Text = text
            toggleText.TextColor3 = Color3.fromRGB(255, 255, 255)
            toggleText.TextSize = 14
            toggleText.TextXAlignment = Enum.TextXAlignment.Left
            toggleButton.Name = "ToggleButton"
            toggleButton.Parent = toggleFrame
            toggleButton.BackgroundTransparency = 1
            toggleButton.Size = UDim2.new(0, 330, 0, 38)
            toggleButton.Text = ""
            toggleDisplay.Name = "ToggleDisplay"
            toggleDisplay.Parent = toggleFrame
            toggleDisplay.Position = UDim2.new(0, 290, 0, 11)
            toggleDisplay.Size = UDim2.new(0, 26, 0, 16)
            -- المربع يشتغل أخضر زمردي أو يطفي رمادي
            toggleDisplay.BackgroundColor3 = toggled and Color3.fromRGB(0, 255, 150) or Color3.fromRGB(45, 45, 50)
            displayCorner.CornerRadius = UDim.new(0, 4)
            displayCorner.Parent = toggleDisplay
            
            toggleButton.MouseButton1Click:Connect(function()
                toggled = not toggled
                if toggled then
                    toggleDisplay.BackgroundColor3 = Color3.fromRGB(0, 255, 150)
                else
                    toggleDisplay.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
                end
                pcall(callback, toggled)
            end)
        end
        return pageElements
    end
    return IceHub
end
